###Add your repo
* Just add the full name of your repo into the following array
* Please copy from the url of your repo instead of typing it
~~~~
$scope.repos = [
  "lianliu/LianBlog",
  "lianliu/SF-Muni",
  "lianliu/lianliu.github.io",
  "lianliu/mean_practice",
  "lianliu/DEBS2016",
  "S1ngS1ng/HammerSpoon",
  "Zhou42/WeatherRadar"
];
~~~~
